#include <bits/stdc++.h>
using namespace std;

const int N = 1e5+9;
int a[N],n;

bitset<N> vis;//记录是否被访问过

void dfs(int dep)//深度(层数)
{
	if(dep==n+1)
	{
		for(int i=1;i<=n;++i)cout<<a[i]<<" \n"[i==n];//当 i 等于 n 时，输出一个换行符 \n，否则输出一个空格
		return;
	}
	for(int i=1;i<=n;++i)
	{
		if(vis[i])continue;

		vis[i]=true;
		a[dep]=i;
		
		dfs(dep+1);
		
		//恢复现场
		a[dep]=0;
		vis[i]=false;
	}
}

int main()
{
	ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
	
	cin>>n;
	
	dfs(1);
	
	return 0;
}
